        bool update_desc(std::string desc_name,float value){

            find_desc.set_desc(desc_name);
            std::vector< std::pair<float, std::string> >::iterator it;
            it = std::find_if( desc.begin(), desc.end(), find_desc );
            if (it == desc.end())
                return false;
            else{
                (*it).first=value;
                return true;
            }
        }
